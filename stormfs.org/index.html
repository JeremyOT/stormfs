mount amazon s3 as a local filesystem: README

$ wget https://github.com/downloads/benlemasurier/stormfs/stormfs-0.01.tar.gz

$ tar -xzvf stormfs-0.01.tar.gz
$ cd stormfs-0.01
$ ./configure
$ make
$ sudo make install

$ sudo cp stormfs.conf.example /etc/stormfs.conf
$ vim /etc/stormfs.conf # setup API access keys

$ mkdir ~/s3
$ sudo mount ~/s3 -o allow_other
$ cd ~/s3 && ls # yay
